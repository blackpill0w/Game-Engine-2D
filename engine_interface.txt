/*
 Load sprite sheet, each sheet has pictures of the same size
 Each line in the spritesheet defines a state (Idle, Walk...) which are just ints
 starting from 0.
*/
sh_id1 = engine.spritesheet_manager.load_spritesheet('/path/to/file1.png', (64, 64));
sh_id2 = engine.spritesheet_manager.load_spritesheet('/path/to/file2.png', (64, 64));

player_id = engine.new_character();
// Make spritesheet active for an entity
engine.bind_spritesheet(player_id, sh_id1);

enum class PlayerAnimationState {
  Idle = 0,
  Walk = 1
};

engine.set_entity_animation_state(player_id, PlayerAnimationState::Walk);
engine.bind(Engine::KeyEvent{key='D', ctrl=true, alt=false}, [&](){
  engine.move_entity(player_id, Vector2{ Horiz=-200, Vert=0 });
});

engine.bind(Engine::Paused, [&]() {
  // code for drawing a pause menu;
});

engine.bind(Engine::MouseEvent{ key="left" }, [&](){
	engine.pause();
});
